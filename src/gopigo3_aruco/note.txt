from ament_index_python.packages import get_package_share_directory
from launch import LaunchDescription
from launch.actions import IncludeLaunchDescription, GroupAction, DeclareLaunchArgument, OpaqueFunction
from launch.launch_description_sources import PythonLaunchDescriptionSource
from launch.substitutions import LaunchConfiguration
from launch_ros.actions import SetRemap ,Node
import xacro





def generate_launch_setup(context, *args, **kwargs):
    
    robot_namespace = LaunchConfiguration('robot_namespace', default='')

    robot_ns_sl = robot_namespace.perform(context) + '/' if robot_namespace.perform(context) != '' else ''

    robot_namespace_arg = DeclareLaunchArgument('robot_namespace', default_value=robot_namespace)

    local_dir  = get_package_share_directory('gopigo3_bringup')
    cam_dir = get_package_share_directory('usb_cam')

    bringup_cmd = IncludeLaunchDescription(
        PythonLaunchDescriptionSource(
            os.path.join(local_dir, 'launch', 'gopigo3_bringup_launch.py')),
            launch_arguments={
                'robot_namespace' : robot_namespace,
            }.items())
    
    robot_cam = IncludeLaunchDescription(
        PythonLaunchDescriptionSource(
            os.path.join(cam_dir, 'launch', 'camera.launch.py')),
        )
    robot_aruco = GroupAction(
        actions=[
            Node(
            package='aruco_opencv',
            executable='aruco_tracker_autostart',
            name='aruco_tracker',
            namespace=robot_namespace,
            output='screen',
            parameters=[{'cam_base_topic' : '/camera1/image_raw',
                         'marker_size': 0.15,
                         }]
            ),
            ])

    
    
    return [
        robot_namespace_arg,
        bringup_cmd,
        robot_cam,
        robot_aruco
    ]

def generate_launch_description():
  return LaunchDescription([
      OpaqueFunction(function=generate_launch_setup)
  ])
